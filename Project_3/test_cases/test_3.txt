

function test_3 z: integer returns integer;
    x: integer is 3;
    y: integer is 18;
begin
    if ((z >= 0) and (z <= 10)) then 
        case z is
            when 0 =>  x + z;
            when 1 =>  y + z;
            when 2 =>  x * y;
            when 3 =>  z - y; 
            when 4 =>  z / x;
            when 5 =>  y + x;
            when 6 =>  y - x;
            when 7 =>  x / y;
            when 8 =>  z ** y;
            when 9 =>  z rem y;
            others =>  y rem x;
        endcase;
    else
        y;
    endif;
end;
